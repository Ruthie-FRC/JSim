46 void applyTo(BallGamepieceSim& sim) const {
47 for (const auto& obstacle : obstacles) {
48 sim.addFieldElement(obstacle.boundary);
49 }
50 for (const auto& goal : goals) {
51 BallGamepieceSim::GoalZone zone{};
52 zone.shape = (goal.shape == GoalStructure::Shape::kSphere)
53 ? BallGamepieceSim::GoalZone::Shape::kSphere
54 : BallGamepieceSim::GoalZone::Shape::kBox;
55 zone.center_m = goal.center_m;
56 zone.half_extents_m = goal.half_extents_m;
57 zone.radius_m = goal.radius_m;
58 switch (goal.accepted_type) {
59 case GoalStructure::AcceptedType::kAny:
60 zone.accept_any_type = true;
61 zone.accepted_type = BallGamepieceSim::GamePieceType::kBall;
62 break;
63 case GoalStructure::AcceptedType::kBall:
64 zone.accepted_type = BallGamepieceSim::GamePieceType::kBall;
65 break;
66 case GoalStructure::AcceptedType::kCustom1:
67 zone.accepted_type = BallGamepieceSim::GamePieceType::kCustom1;
68 break;
69 case GoalStructure::AcceptedType::kCustom2:
70 zone.accepted_type = BallGamepieceSim::GamePieceType::kCustom2;
71 break;
72 case GoalStructure::AcceptedType::kCustom3:
73 zone.accepted_type = BallGamepieceSim::GamePieceType::kCustom3;
74 break;
75 case GoalStructure::AcceptedType::kCustom4:
76 zone.accepted_type = BallGamepieceSim::GamePieceType::kCustom4;
77 break;
78 }
79 zone.require_positive_vertical_velocity =
80 goal.require_positive_vertical_velocity;
81 zone.custom_velocity_validator = goal.custom_velocity_validator;
82 sim.addGoalZone(zone);
83 }
84 }

216 void simulationPeriodic() {
217 const int subticks = std::max(1, timings_.simulation_subticks_per_period);
218 const double dt = timings_.robot_period_s / static_cast<double>(subticks);
219
220 for (int sub_tick = 0; sub_tick < subticks; ++sub_tick) {
221 gamepiece_sim_.step(dt);
222
223 for (auto& intake : intake_simulations_) {
224 intake.update(gamepiece_sim_);
225 }
226
227 for (auto& custom_sim : custom_simulations_) {
228 custom_sim(sub_tick, *this);
229 }
230 }
